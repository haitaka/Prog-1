    \begin{frame}[fragile]{Условный оператор}
        \begin{onlyenv}<1>
            \begin{minted}{c}
                if (x == 42) {
                    printf("x is 42");
                }
            \end{minted}
        \end{onlyenv}
        \begin{onlyenv}<2>
            \begin{minted}{c}
                if (x == 42) {
                    printf("x is 42");
                } else {
                    printf("x is not 42");
                }
            \end{minted}
        \end{onlyenv}
        \begin{onlyenv}<3>
            \begin{minted}{c}
                if (x == 42) {
                    printf("x is 42");
                } else {
                    if (x == 37) {
                        printf("x is 37");
                    } else {
                        printf("x is not 42");
                    }
                }
            \end{minted}
        \end{onlyenv}
        \begin{onlyenv}<4>
            \begin{minted}{c}
                if (x == 42) {
                    printf("x is 42");
                } else if (x == 37) {
                    printf("x is 37");
                } else {
                    printf("x is not 42");
                }
            \end{minted}
        \end{onlyenv}
    \end{frame}

    \begin{frame}[fragile]{Цикл}
        \begin{minted}{c}
            while (x > 10) {
                x -= 1;
            }
        \end{minted}
    \end{frame}

    \begin{frame}[fragile]{Цикл}
        \begin{minted}{c}
            do {
                x -= 1;
            } while (x > 10);
        \end{minted}
    \end{frame}

    \begin{frame}[fragile]{Цикл}
        \begin{columns}
            \begin{column}{0.6\textwidth}
                \begin{minted}{c}
                    for (int i = 0; i < 10; ++i) {
                        printf("hi");
                    }
                \end{minted}
            \end{column}
            \begin{column}{0.4\textwidth}<2->
                \begin{onlyenv}<2>
                    \begin{minted}{c}
                        int i = 0;
                        while (i < 10) {
                            printf("hi");
                            ++i;
                        }
                    \end{minted}
                \end{onlyenv}
                \begin{onlyenv}<3>
                \begin{minted}{c}
                        int i = 0;
                        while (1) {
                            if (i >= 10) {
                                break;
                            }
                            printf("hi");
                            ++i;
                        }
                    \end{minted}
                \end{onlyenv}
            \end{column}
        \end{columns}
    \end{frame}

    \begin{frame}{Цикл}
        \begin{itemize}
            \item \code{break} --- прервать цикл
            \item \code{continue} --- прервать текущую итерацию цикла
        \end{itemize}
    \end{frame}

    \begin{frame}[fragile]{Switch}
        \begin{columns}
            \begin{column}{0.5\textwidth}
                \begin{minted}{c}
                    switch (x) {
                        case 1:
                            printf("first");
                        case 2:
                            printf("second");
                            break;
                        case 3:
                            printf("third");
                        default:
                            printf("default");
                    }
                \end{minted}
            \end{column}
            \begin{column}{0.5\textwidth}<2>
                \begin{minted}{c}
                    if (x == 1) {
                        printf("first");
                        printf("second");
                    } else if (x == 2) {
                        printf("second");
                    } else if (x == 3) {
                        printf("third");
                        printf("default");
                    } else {
                        printf("default");
                    }
                \end{minted}
            \end{column}
        \end{columns}
    \end{frame}

\end{document}
